            tabs.forEach(id => {
                const btn = document.getElementById(`btn-${id}`);
                const content = document.getElementById(`content-${id}`);
                
                if (id === tabId) {
                    // Activate clicked tab
                    btn.classList.remove('tab-inactive');
                    btn.classList.add('tab-active');
                    content.classList.remove('hidden');
                } else {
                    // Deactivate others
                    btn.classList.remove('tab-active');
                    btn.classList.add('tab-inactive');
                    content.classList.add('hidden');
                }
            });

            // Initialize charts only when Matching tab is clicked (to ensure canvas is visible)
            if (tabId === 'matching' && !appState.chartsInitialized) {
                initMatchChart();
                appState.chartsInitialized = true;
            }
        }

        // Initialize Chart.js Visualizations
        function initCharts() {
            // Setup Performance Bar Chart (Always visible section)
            const ctxPerf = document.getElementById('performanceChart').getContext('2d');
            new Chart(ctxPerf, {
                type: 'bar',
                data: {
                    labels: ['Manual Screening', 'AI Batch Processing'],
                    datasets: [{
                        label: 'Time Required (Hours)',
                        data: [50, 0.25], // 1000 * 3mins = 3000mins = 50 hours vs ~15 mins batch
                        backgroundColor: [
                            '#94a3b8', // Slate 400
                            '#4f46e5'  // Indigo 600
                        ],
                        borderRadius: 6
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { display: false },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    return context.raw + ' Hours';
                                }
                            }
                        }
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            title: { display: true, text: 'Hours to Process 1,000 CVs' }
                        }
                    }
                }
            });
        }

        // Setup Match Score Doughnut Chart (Rendered when tab opens)
        function initMatchChart() {
            const ctxMatch = document.getElementById('matchScoreChart').getContext('2d');
            new Chart(ctxMatch, {
                type: 'doughnut',
                data: {
                    labels: ['Match', 'Gap'],
                    datasets: [{
                        data: [75, 25],
                        backgroundColor: ['#f59e0b', '#f1f5f9'], // Amber 500, Slate 100
                        borderWidth: 0,
                        cutout: '80%'
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    animation: { animateScale: true, animateRotate: true },
                    plugins: {
                        legend: { display: false },
                        tooltip: { enabled: false }
                    }
                },
                plugins: [{
                    id: 'textCenter',
                    beforeDraw: function(chart) {
                        var width = chart.width,
                            height = chart.height,
                            ctx = chart.ctx;

                        ctx.restore();
                        var fontSize = (height / 80).toFixed(2);
                        ctx.font = "bold " + fontSize + "em Inter, sans-serif";
                        ctx.textBaseline = "middle";
                        ctx.fillStyle = "#0f172a";

                        var text = "75%",
                            textX = Math.round((width - ctx.measureText(text).width) / 2),
                            textY = height / 2;

                        ctx.fillText(text, textX, textY);
                        ctx.save();
                    }
                }]
            });
        }

        // Initialize globally visible charts on DOM load
        document.addEventListener('DOMContentLoaded', () => {
            initCharts();
        });
